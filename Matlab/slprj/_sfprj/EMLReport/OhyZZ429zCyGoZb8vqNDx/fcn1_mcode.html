1function xdot = fcn(x,u)
 2%#codegen
 3
 4% extract states and inputs
 5v=x(4:6);
 6
 7r=x(7);
 8p=x(8);
 9y=x(9);
10
11w1=x(10);
12w2=x(11);
13w3=x(12);
14
15T=sum(u);
16T1=u(1);
17T2=u(2);
18T3=u(3);
19T4=u(4);
20
21%constants
22g=9.81;
23m=0.027;
24b=1e-3;
25k=b;
26d=0.1;
27Ix = 1.395e-5;
28Iy = 1.436e-5;
29Iz = 2.173e-5;
30
31invJ=inv([Ix,0,0;
32            0,Iy,0;
33            0,0,Iz]);
34invBigT=inv([1,0,-p;
35            0,1,r;
36            0,-r,1]);
37
38As=[0 -d 0 d;
39    d 0 -d 0;
40    k/b -k/b k/b -k/b];
41
42%equations
43xyzdot=v;
44
45vdot=[-p*y*T/m;
46        p*T/m;
47        g-T/m];
48    
49rpydot=invBigT*[w1;w2;w3];
50
51wdot=invJ*As*[T1;T2;T3;T4] - invJ*[(Iy-Iz)*w2*w3;
52                                         (Iz-Ix)*w1*w3;
53                                         (Ix-Iy)*w1*w2];
54
55
56xdot = [xyzdot; vdot; rpydot; wdot];
